# 실시간 모니터링 시스템 - 실제 서비스 구현 가이드

## 개요
현재 목업(Mock) 모드로 구현된 실시간 모니터링 시스템을 실제 서비스로 전환하기 위한 API 명세서 및 데이터 요구사항

## 1. 실시간 비디오 스트리밍

### 1.1 비디오 스트림 API
**목적**: 카메라로부터 실시간 비디오 피드를 전송

#### WebSocket 연결 (권장)
```
WebSocket URL: ws://[server]/api/v1/video/stream?camera_id=CAM:01
```

**연결 시 전송 데이터**:
```json
{
  "camera_id": "CAM:01",
  "quality": "1080p",
  "fps": 30
}
```

**수신 데이터** (바이너리 프레임 또는 Base64):
```json
{
  "frame": "base64_encoded_image",
  "timestamp": "2026-01-26T10:30:45.123Z",
  "metadata": {
    "resolution": "1920x1080",
    "fps": 30,
    "latency_ms": 45
  }
}
```

#### 대안: HTTP 스트리밍
```
GET /api/v1/video/stream/{camera_id}
Response: multipart/x-mixed-replace; boundary=frame
```

## 6. 프론트엔드 구현 가이드

### 6.1 WebSocket 연결 관리

```javascript
// LiveVideoView.vue에 추가할 WebSocket 연결 코드

import { ref, onMounted, onUnmounted } from 'vue'

// WebSocket 연결
let vehiclePositionWs = null
let vehicleStatusWs = null
let sensorStatusWs = null
let scanRealtimeWs = null

const connectWebSockets = () => {
  const baseWsUrl = import.meta.env.VITE_WS_URL || 'ws://127.0.0.1:8000'

  // 1. 차량 위치 WebSocket
  vehiclePositionWs = new WebSocket(`${baseWsUrl}/api/v1/vehicle/position`)

  vehiclePositionWs.onmessage = (event) => {
    const data = JSON.parse(event.data)

    // 차량 위치 업데이트
    vehicle.value.x = data.position.x
    vehicle.value.y = data.position.y
    vehicle.value.angle = data.orientation.angle

    // 경로 추가
    vehiclePath.value.push({
      x: data.position.x,
      y: data.position.y
    })

    if (vehiclePath.value.length > 100) {
      vehiclePath.value.shift()
    }

    // 속도 업데이트
    vehicleStatus.value.speed = data.velocity.speed

    updateViewBox()
  }

  vehiclePositionWs.onerror = (error) => {
    console.error('Vehicle Position WebSocket Error:', error)
  }

  vehiclePositionWs.onclose = () => {
    console.log('Vehicle Position WebSocket Closed')
    // 재연결 로직
    setTimeout(connectWebSockets, 5000)
  }

  // 2. 차량 상태 WebSocket
  vehicleStatusWs = new WebSocket(`${baseWsUrl}/api/v1/vehicle/status`)

  vehicleStatusWs.onmessage = (event) => {
    const data = JSON.parse(event.data)

    vehicleStatus.value = {
      mode: data.mode,
      speed: data.speed.current,
      battery: data.battery.level,
      distanceToTarget: data.navigation.distance_to_target,
      eta: data.navigation.eta_seconds
    }
  }

  // 3. 센서 상태 WebSocket
  sensorStatusWs = new WebSocket(`${baseWsUrl}/api/v1/sensors/status`)

  sensorStatusWs.onmessage = (event) => {
    const data = JSON.parse(event.data)

    sensors.value = data.sensors.map(sensor => ({
      name: sensor.name,
      status: sensor.status,
      value: sensor.data.resolution || sensor.data.accuracy || sensor.data.field_of_view || '정상'
    }))
  }

  // 4. 스캔 실시간 WebSocket
  scanRealtimeWs = new WebSocket(`${baseWsUrl}/api/v1/waybills/scan/realtime`)

  scanRealtimeWs.onmessage = (event) => {
    const data = JSON.parse(event.data)

    if (data.status === 'RECOGNIZED' || data.status === 'READY') {
      currentScan.value = {
        waybill_id: data.waybill_id,
        tracking_number: data.recognition_data.tracking_number,
        destination: data.recognition_data.destination,
        status: data.status,
        confidence_score: data.recognition_data.confidence_score,
        created_at: data.timestamp
      }

      // 히스토리에 추가
      scanHistory.value.unshift(currentScan.value)
      if (scanHistory.value.length > 10) {
        scanHistory.value.pop()
      }
    }
  }
}

const disconnectWebSockets = () => {
  if (vehiclePositionWs) vehiclePositionWs.close()
  if (vehicleStatusWs) vehicleStatusWs.close()
  if (sensorStatusWs) sensorStatusWs.close()
  if (scanRealtimeWs) scanRealtimeWs.close()
}

onMounted(() => {
  // 기존 목업 시뮬레이션 제거 또는 조건부 실행
  if (!import.meta.env.VITE_MOCK_MODE) {
    connectWebSockets()
  } else {
    // 목업 모드에서는 기존 시뮬레이션 실행
    simulationInterval = setInterval(simulateVehicleMovement, 50)
  }
})

onUnmounted(() => {
  disconnectWebSockets()
  if (simulationInterval) {
    clearInterval(simulationInterval)
  }
})
```

```typescript
// types/monitoring.ts

export interface VehiclePosition {
  vehicle_id: string
  position: {
    x: number
    y: number
    z: number
    coordinate_system: 'local' | 'gps'
  }
  orientation: {
    angle: number
    quaternion?: {
      x: number
      y: number
      z: number
      w: number
    }
  }
  velocity: {
    speed: number
    direction: number
  }
  timestamp: string
}

export interface VehicleStatus {
  vehicle_id: string
  mode: 'AUTO' | 'MANUAL' | 'IDLE' | 'EMERGENCY_STOP'
  battery: {
    level: number
    voltage: number
    current: number
    estimated_runtime: number
    is_charging: boolean
  }
  speed: {
    current: number
    max: number
    limit: number
  }
  navigation: {
    current_waypoint: string
    next_waypoint: string
    distance_to_target: number
    eta_seconds: number
    route_completion: number
  }
  health: {
    status: 'normal' | 'warning' | 'error' | 'critical'
    warnings: string[]
    errors: string[]
  }
  timestamp: string
}

export interface SensorData {
  vehicle_id: string
  sensors: Array<{
    name: string
    type: string
    status: 'ok' | 'warning' | 'error' | 'offline'
    health: number
    data: Record<string, any>
    last_update: string
  }>
  timestamp: string
}

export interface WaybillScanResult {
  session_id: string
  waybill_id: string
  recognition_data: {
    tracking_number: string
    destination: string
    recipient: string
    confidence_score: number
  }
  status: 'SCANNING' | 'RECOGNIZED' | 'READY' | 'MOVING' | 'COMPLETED'
  timestamp: string
}
```
